```python
>>> import torch
>>> a = torch.ones(3)
>>> a
tensor([1., 1., 1.])
>>> b = 10
>>> a + b
tensor([11., 11., 11.])

>>> c=numpy.ones((2,3))

       [1., 1., 1.]])
#torch可以直接传进
>>> d=torch.ones(2,3)

        [1., 1., 1.]])
```
